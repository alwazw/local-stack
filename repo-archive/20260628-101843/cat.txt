    STATUS2=$(cat /sys/class/net/"$NIC"/operstate | tr '[:lower:]' '[:upper:]')

    # Get IP address (if any)
    IP=$(ip -brief addr show "$NIC" | awk '{print $3}' | head -n 1)
    [ -z "$IP" ] && IP="NO-IP"

    # Colorize Status 1 (Admin)
    [ "$STATUS1" = "UP" ] && S1="\e[32mUP\e[0m" || S1="\e[31mDOWN\e[0m"

    # Colorize Status 2 (Operational)
    [ "$STATUS2" = "UP" ] && S2="\e[32mUP\e[0m" || S2="\e[31mDOWN\e[0m"

    # Print the line
    echo -e "$NIC\t $S1\t $S2\t$IP"
done
}


## ipb
ipb() {
    local G='\033[0;32m' R='\033[0;31m' Y='\033[1;33m' C='\033[1;36m' NC='\033[0m'
    local B='\033[1m'
    for nic in $(ls /sys/class/net | grep -vE 'lo|veth|fw|docker|br-|vmbr|bonding'); do
        [[ ! -d "/sys/class/net/$nic" ]] && continue
        local status=$(cat "/sys/class/net/$nic/operstate" 2>/dev/null || echo "unknown")
        local mac=$(cat "/sys/class/net/$nic/address" 2>/dev/null)
        local bridge=$(ip link show "$nic" | grep -oP 'master \K[^ ]+')
        local ip_line=$(ip -4 -br addr show dev "${bridge:-$nic}" 2>/dev/null)
        local ip_addr=$(echo "$ip_line" | awk '{print $3}' | head -n1)
        local gw=$(ip route show default dev "${bridge:-$nic}" 2>/dev/null | awk '{print $3}')

        local dstat=$([ "$status" = "up" ] && echo "${G}up" || echo "${R}down")
        local ip_disp=$([ -z "$ip_addr" ] && echo "${R}unassigned" || echo "${C}$ip_addr${NC}")
        echo -e "\n${B}$nic ${dstat}${NC}"
        echo -e "  MAC: $mac"
        echo -e "  IP : $ip_disp"
        echo -e "  GW : ${gw:-none}"
    done
    echo
}

## venv
venv() {
    local VENV_DIR=".venv"
    if [[ "$1" == "clean" || "$1" == "--clean" ]]; then
        if [[ -n "$VIRTUAL_ENV" ]]; then deactivate; fi
        rm -rf "$VENV_DIR"
        echo "🗑️  Removed $VENV_DIR"
        return
    fi
    if [[ ! -d "$VENV_DIR" ]]; then
        echo "🚀 Creating virtual environment in $VENV_DIR"
        python3 -m venv "$VENV_DIR"
    fi
    source "$VENV_DIR/bin/activate"
    if [[ -f requirements.txt ]]; then
        echo "📦 Installing Python dependencies"
        pip install --upgrade pip && pip install -r requirements.txt
    fi
}

dps() {
    # 1. Define ANSI Color Codes
    local CYAN='\033[0;36m'
    local GREEN='\033[1;32m'
    local RED='\033[1;31m'
    local YELLOW='\033[0;33m'
    local NC='\033[0m' # No Color

    # 2. Define the structural Go template format for Docker
    local FORMAT="table {{.ID}}\t{{.Names}}\t{{.CreatedAt}}\t{{.Status}}\t{{.Ports}}"

    echo -e "${CYAN}Mapping Active Container Topology...${NC}\n"

    # 3. Execute Docker ps, clean up port strings, and inject conditional coloring
    docker ps -a --format "$FORMAT" | awk -v grn="$GREEN" -v red="$RED" -v yel="$YELLOW" -v nc="$NC" '
    NR==1 { print; next } # Print the table header untouched
    {
        # Clean up the chaotic multi-ip port bindings string
        # Replaces "0.0.0.0:80->80/tcp, [::]:80->80/tcp" with a clean "80->80/tcp"
        gsub(/[0-9.]+:|\[::\]:/, "", $NF);

        # Look at the Status column fields to determine runtime health color
        if ($0 ~ /Up [0-9]+/) {
            # Container is actively running smoothly
            match($0, /Up [^\t]+/);
            status=substr($0, RSTART, RLENGTH);
            gsub(status, grn status nc);
        } else if ($0 ~ /Exited \([0-9]+\)/) {
            # Container has dropped offline or broken
            match($0, /Exited \([0-9]+\) [^\t]+/);
            status=substr($0, RSTART, RLENGTH);
            gsub(status, red status nc);
        } else {
            # Container is in an intermediate state (e.g. Created, Restarting)
            match($0, /(Created|Restarting) [^\t]+/);
            status=substr($0, RSTART, RLENGTH);
            gsub(status, yel status nc);
        }
        print;
    }' | column -t -s $'\t' # Use system tab-stops to keep tables perfectly justified
}

dps2() {
    # 1. Define ANSI Color Codes
    local CYAN='\033[0;36m'
    local GREEN='\033[1;32m'
    local RED='\033[1;31m'
    local YELLOW='\033[0;33m'
    local NC='\033[0m'
    local FAINT='\033[2m'

    # Handle Runtime CLI Filtering Arguments (e.g., dps up, dps down)
    local FILTER_ARG=$(echo "$1" | tr '[:upper:]' '[:lower:]')

    local FORMAT='{"id":"{{.ID}}","name":"{{.Names}}","created":"{{.CreatedAt}}","status":"{{.Status}}","ports":"{{.Ports}}"}'

    echo -e "${CYAN}Analyzing Complete Deployment Topology (Active & Terminated)...${NC}\n"

    # Master Layout Header Block
    printf "%-14s %-18s %-18s %-14s %-12s %-16s %-20s %s\n" \
        "CONTAINER ID" "NAMES" "CREATED AT" "STATE" "HEALTH" "DURATION" "BASE URL" "PORTS"

    docker ps -a --format "$FORMAT" | awk -v grn="$GREEN" -v red="$RED" -v yel="$YELLOW" -v fnt="$FAINT" -v nc="$NC" -v filter="$FILTER_ARG" '
    BEGIN {
        # Initialize running tally buckets for the structural footer summary
        up_count = 0;
        down_count = 0;
        failed_count = 0;
        displayed_count = 0;
    }
    {
        # --- 1. EXTRACT JSON ATTRIBUTES ---
        match($0, /"id":"[^"]*"/); id = substr($0, RSTART+6, RLENGTH-7);
        match($0, /"name":"[^"]*"/); name = substr($0, RSTART+8, RLENGTH-9);
        match($0, /"created":"[^"]*"/); created = substr($0, RSTART+11, RLENGTH-12);
        match($0, /"status":"[^"]*"/); status_raw = substr($0, RSTART+10, RLENGTH-11);
        match($0, /"ports":"[^"]*"/); ports_raw = substr($0, RSTART+9, RLENGTH-10);

        # --- 2. CLEAN CREATED AT DATE ---
        split(created, date_parts, " ");
        created_clean = date_parts[1] " " substr(date_parts[2], 1, 5);

        # --- 3. PARSE STATUS METRICS & INCREMENT TALLIES ---
        if (status_raw ~ /^Up/) {
            state_raw = "UP"; state_word = grn state_raw nc;
            up_count++;
            is_up = 1; is_failed = 0;
        } else if (status_raw ~ /Exited \(0\)/) {
            state_raw = "DOWN"; state_word = red state_raw nc;
            down_count++;
            is_up = 0; is_failed = 0;
        } else if (status_raw ~ /Exited \([1-9][0-9]*\)/) {
            state_raw = "DOWN FAILED"; state_word = red state_raw nc;
            failed_count++;
            is_up = 0; is_failed = 1;
        } else {
            state_raw = "PENDING"; state_word = yel state_raw nc;
            is_up = 0; is_failed = 0;
        }

        if (status_raw ~ /\(healthy\)/) {
            health_raw = "HEALTHY"; health_word = grn health_raw nc;
        } else if (status_raw ~ /\(unhealthy\)/) {
            health_raw = "UNHEALTHY"; health_word = red health_raw nc;
        } else if (status_raw ~ /\(health: starting\)/) {
            health_raw = "STARTING"; health_word = yel health_raw nc;
        } else {
            health_raw = ""; health_word = "";
        }

        t_str = status_raw;
        gsub(/^Up |^Exited \([0-9]+\) /, "", t_str);
        gsub(/ \(healthy\)|\(unhealthy\)|\(health: starting\)/, "", t_str);
        duration = t_str;

        # --- 4. EVALUATE FILTER ARGUMENTS ---
        if (filter == "up" && !is_up) next;
        if (filter == "down" && is_up) next;
        if (filter == "failed" && !is_failed) next;

        displayed_count++;

        # --- 5. MAP BASE URL NETWORK TOPOLOGY ---
        base_url = "localhost";
        if (name ~ /traefik/)          base_url = "domain.com";
        else if (name ~ /vaultwarden/) base_url = "://domain.com";
        else if (name ~ /openwebui/)   base_url = "://domain.com";
        else if (name ~ /authentik/)   base_url = "://domain.com";

        # --- 6. NORMALIZE & DEDUPLICATE PORTS ---
        port_count = 0;
        delete port_arr;
        delete seen_ports;

        if (ports_raw != "") {
            n_ports = split(ports_raw, raw_arr, ", ");
            for (i = 1; i <= n_ports; i++) {
                p = raw_arr[i];
                proto = "TCP";
                if (p ~ /\/udp/) proto = "UDP";

                if (p ~ /->/) {
                    gsub(/[0-9.]+:|\[::\]:/, "", p);
                    match(p, /[0-9]+/);
                    host_port = substr(p, RSTART, RLENGTH);

                    if (host_port != "") {
                        port_str = proto ": " host_port;
                        if (!seen_ports[port_str]++) {
                            port_count++;
                            port_arr[port_count] = port_str;
                        }
                    }
                } else {
                    match(p, /[0-9]+/);
                    container_port = substr(p, RSTART, RLENGTH);
                    if (container_port != "") {
                        port_str = proto ": " container_port;
                        if (!seen_ports[port_str]++) {
                            port_count++;
                            port_arr[port_count] = port_str;
                        }
                    }
                }
            }
        }

        # --- 7. PRINT BALANCED ROW DATA ---
        state_pad  = 14 + (length(state_word) - length(state_raw));
        health_pad = 12 + (length(health_word) - length(health_raw));

        first_port = (port_count > 0) ? port_arr[1] : "";

        printf "%-14s %-18s %-18s %-"state_pad"s %-"health_pad"s %-16s %-20s %s\n", \
            id, name, created_clean, state_word, health_word, duration, base_url, first_port;

        if (port_count > 1) {
            for (j = 2; j <= port_count; j++) {
                printf "%-14s %-18s %-18s %-14s %-12s %-16s %-20s %s\n", \
                    "", "", "", "", "", "", "", port_arr[j];
            }
        }

        # --- 8. INJECT EXTENDED 130-CHARACTER SEPARATOR ---
        printf "%s----------------------------------------------------------------------------------------------------------------------------------\n%s", fnt, nc;
    }
    END {
        # --- 9. RENDER STRUCTURAL FOOTER METRICS ---
        printf "\n%s[ SYSTEM ENGINE SUMMARY ]%s\n", yel, nc;
        printf "Total Discovered: %-4d | ", (up_count + down_count + failed_count);
        printf "Active (%sUP%s): %-4d | ", grn, nc, up_count;
        printf "Stopped (%sDOWN%s): %-4d | ", red, nc, down_count;
        printf "Crashed (%sFAILED%s): %-4d\n", red, nc, failed_count;

        if (filter != "") {
            printf "%sActive Filter Applied: \"%s\" (%d records displayed)%s\n", yel, filter, displayed_count, nc;
        }
    }'
}
